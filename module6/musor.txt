     class Treug
    {
        public static int a, b, c;

        static int P_Treug()
        {
            var P = a + b + c; 
            return P;
        }



        static double S_Treug()
        {

            var p = Treug.P_Treug()/2;
            var S = Math.Sqrt(p * (p - a) * (p - b) * (p - c));  
            return S;
        }


    }
